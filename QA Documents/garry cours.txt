 STRUC -> defines shape of data. (doesnt compil anything) 
 STRUC Vec 2
      x resw 1
      y resw 1 
ENDSTRUC

section .data

position: ISTRUC Vec2
      at .x, dw 1 
      at .y, dw 2
mov ax, [position + Vec2.x]
mov bx, [position + Vec2.y] 
mov sI, [velocity]
mov ax, [sI + Vec2.x]
mov bx, [sI + Vec2.y]

section .bss 
velocity resb Vec2_size
